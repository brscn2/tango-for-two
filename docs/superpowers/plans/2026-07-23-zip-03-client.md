- [ ] **Step 3: Implement using `canExtendPath` from `@tango/shared`**

```ts
import { canExtendPath, type PublicZipPuzzle, type ZipCoord } from '@tango/shared';
import type { ZipPuzzle } from '@tango/shared';

function asPuzzle(p: PublicZipPuzzle): ZipPuzzle {
  return { ...p, waypoints: p.waypoints, walls: p.walls };
}

export function extendPathFromPaint(
  path: ZipCoord[],
  cell: ZipCoord,
  puzzle: PublicZipPuzzle,
): ZipCoord[] {
  if (path.length === 0) {
    const start = puzzle.waypoints.find((w) => w.n === 1);
    if (!start || start.r !== cell.r || start.c !== cell.c) return path;
    return [cell];
  }
  // Retract: clicking/painting onto the previous cell pops tip (optional LinkedIn-like undo)
  const prev = path[path.length - 2];
  if (prev && prev.r === cell.r && prev.c === cell.c) {
    return path.slice(0, -1);
  }
  if (!canExtendPath(path, cell, asPuzzle(puzzle))) return path;
  return [...path, cell];
}

export function clearPath(): ZipCoord[] {
  return [];
}
```

- [ ] **Step 2: Handlers**

```ts
socket.on('roomState', (s) => {
  set({
    code: s.code,
    players: s.players,
    scores: s.scores,
    match: s.match,
    symbols: s.symbols,
    myBoard: s.match?.gameType === 'tango'
      ? (get().myBoard ?? (s.match.puzzle ? cloneGrid(s.match.puzzle.clues) : null))
      : null,
    myPath: s.match?.gameType === 'zip' ? (get().myPath ?? []) : null,
  });
});

socket.on('matchStarted', (m) => {
  if (m.gameType === 'zip') {
    set({ match: m, myPath: [], myBoard: null, opponentFilled: 0 });
  } else {
    set({ match: m, myBoard: cloneGrid(m.puzzle!.clues), myPath: null, opponentFilled: 0 });
  }
});

socket.on('zipPathSync', ({ path, opponentFilled }) => {
  set({ myPath: path, opponentFilled });
});

socket.on('matchWon', (p) => {
  set((st) => ({
    scores: p.scores,
    match: st.match ? { ...st.match, status: 'won', winnerSlot: p.winnerSlot } : null,
  }));
});

startMatch(mode, difficulty, size = 6, gameType = 'tango') {
  get().socket!.emit('startMatch', { mode, difficulty, size, gameType });
},

setZipPath(path) {
  set({ myPath: path });
  get().socket!.emit('zipPathUpdate', { path });
},

clearZipPath() {
  get().setZipPath([]);
},
```
