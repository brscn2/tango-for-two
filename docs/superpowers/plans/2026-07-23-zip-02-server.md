# Zip Server & Protocol Implementation Plan (Plan 2 of 3)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the Socket.IO protocol and `RoomManager` so rooms can start Zip Race matches, sync paths, detect Zip wins, and persist separate Tango vs Zip score tallies.

**Architecture:** Add `gameType` to match start/state. Zip matches store `zipPuzzle`, `solutionPath`, and per-slot `paths`. Path sync uses full snapshots (`zipPathUpdate`). Scores become `{ tango: ScoreEntry[]; zip: ScoreEntry[] }`. Tango paths stay unchanged.

**Tech Stack:** TypeScript, Socket.IO, Vitest, MongoDB/memory DB.

**Spec:** `docs/superpowers/specs/2026-07-23-linkedin-zip-design.md`  
**Depends on:** Plan 1 (`generateZipPuzzle`, `isValidZipSolution`, Zip types).

startMatch(
  code: string,
  mode: Mode,
  difficulty: Difficulty,
  size: number = 6,
  gameType: GameType = 'tango',
): MatchState {
  const room = this.rooms.get(code);
  if (!room) throw new Error('Room not found');

  if (gameType === 'zip') {
    if (mode !== 'race') throw new Error('Zip only supports race mode');
    if (size !== 6 && size !== 7 && size !== 8) throw new Error('Invalid Zip size');
    const { puzzle, solutionPath } = this.zipGen(size as 6 | 7 | 8, difficulty);

      mode: 'race',

        walls: puzzle.walls,

      paths: [[], []],
      startedAt: Date.now(),
      status: 'playing',
      winnerSlot: null,
    };

    return toMatchState(match);

- [ ] **Step 4: `applyZipPath`**

```ts
applyZipPath(code: string, slot: Slot, path: ZipCoord[]): ApplyResult {
  const room = this.rooms.get(code);
  const match = room?.match;
  if (!match || match.status !== 'playing' || match.gameType !== 'zip' || !match.zipPuzzle) {
    return {};
  }
  const puzzle: ZipPuzzle = { ...match.zipPuzzle };
  const start = puzzle.waypoints.find((w) => w.n === 1);
  if (!start) return {};

  // Empty path = clear
  if (path.length === 0) {
    match.paths![slot] = [];
    return {
      progress: { slot, filled: 0, total: puzzle.size * puzzle.size },
    };
  }

  // Validate full snapshot from scratch
  if (path[0].r !== start.r || path[0].c !== start.c) return {};
  const built: ZipCoord[] = [path[0]];
  for (let i = 1; i < path.length; i++) {
    if (!canExtendPath(built, path[i], puzzle)) return {};
    built.push(path[i]);
  }

  match.paths![slot] = built;
  const total = puzzle.size * puzzle.size;
  const result: ApplyResult = {
    progress: { slot, filled: built.length, total },
  };
  if (isValidZipSolution(built, puzzle)) {
    match.status = 'won';
    match.winnerSlot = slot;
    const timeMs = Date.now() - match.startedAt;
    result.won = {
      winnerSlot: slot,
      timeMs,
      scores: this.db.recordResult(code, 'zip', slot, timeMs),
    };
  }
  return result;
}
```
